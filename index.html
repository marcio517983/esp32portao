#include #include #include #include // --- PINOS --- #define RELE_PIN 5 #define RELE_AUTO_PIN 4 #define RELE_ALERTA_PIN 15 #define RELE_CONFIG_PIN 23 #define SENSOR_ABERTO_PIN 18 #define SENSOR_FECHADO_PIN 19 #define BOTAO_RESET_WIFI 13 // --- VARIÁVEIS MQTT --- char mqtt_server[50] = "e6609cbe84ae4e2b8eda83c3164e7900.s1.eu.hivemq.cloud"; char mqtt_user[20] = "esp32"; char mqtt_pass[20] = "517983lgA"; const int mqtt_port = 8883; const char* topic_comando = "esp32/esp32_01/comando"; const char* topic_sensores = "esp32/esp32_01/sensores"; // --- VARIÁVEIS DE CONTROLE --- unsigned long tempoPulso = 0; bool relePulsoAtivo = false; bool releAutoEstado = false; bool releConfigEstado = false; String ultimoEstado = ""; unsigned long tempoInicioAberto = 0; bool monitorandoAlerta = false; const unsigned long LIMITE_ALERTA = 30000; unsigned long tempoBotaoPressionado = 0; WiFiClientSecure espClient; PubSubClient client(espClient); // --- HTML E CSS CUSTOMIZADO (Identidade CONTROWS) --- const char* custom_html_head = R"=====( )====="; void callback(char* topic, byte* payload, unsigned int length) { String msg = ""; for (int i = 0; i < length; i++) msg += (char)payload[i]; if (msg == "PULSO") { digitalWrite(RELE_PIN, LOW); tempoPulso = millis(); relePulsoAtivo = true; } if (msg == "AUTO") { releAutoEstado = !releAutoEstado; digitalWrite(RELE_AUTO_PIN, releAutoEstado ? LOW : HIGH); } if (msg == "CONFIG") { releConfigEstado = !releConfigEstado; digitalWrite(RELE_CONFIG_PIN, releConfigEstado ? LOW : HIGH); } } void reconnect() { while (!client.connected()) { String clientId = "CONTROWS-" + String(random(0xffff), HEX); if (client.connect(clientId.c_str(), mqtt_user, mqtt_pass)) { client.subscribe(topic_comando); } else { delay(5000); } } } void setup() { Serial.begin(115200); pinMode(RELE_PIN, OUTPUT); pinMode(RELE_AUTO_PIN, OUTPUT); pinMode(RELE_ALERTA_PIN, OUTPUT); pinMode(RELE_CONFIG_PIN, OUTPUT); pinMode(BOTAO_RESET_WIFI, INPUT_PULLUP); digitalWrite(RELE_PIN, HIGH); digitalWrite(RELE_AUTO_PIN, HIGH); digitalWrite(RELE_ALERTA_PIN, HIGH); digitalWrite(RELE_CONFIG_PIN, HIGH); pinMode(SENSOR_ABERTO_PIN, INPUT_PULLUP); pinMode(SENSOR_FECHADO_PIN, INPUT_PULLUP); WiFiManager wm; // Injeta o novo visual no portal wm.setCustomHeadElement(custom_html_head); // Criação dos parâmetros MQTT para o portal WiFiManagerParameter custom_mqtt_server("server", "Broker HiveMQ", mqtt_server, 50); WiFiManagerParameter custom_mqtt_user("user", "Usuário MQTT", mqtt_user, 20); WiFiManagerParameter custom_mqtt_pass("pass", "Senha MQTT", mqtt_pass, 20); wm.addParameter(&custom_mqtt_server); wm.addParameter(&custom_mqtt_user); wm.addParameter(&custom_mqtt_pass); wm.setCaptivePortalEnable(true); // O nome da rede que aparece no Wi-Fi if (!wm.autoConnect("CONTROWS_CONFIG_GATE")) { delay(3000); ESP.restart(); } // Captura os valores digitados no portal strcpy(mqtt_server, custom_mqtt_server.getValue()); strcpy(mqtt_user, custom_mqtt_user.getValue()); strcpy(mqtt_pass, custom_mqtt_pass.getValue()); espClient.setInsecure(); client.setServer(mqtt_server, mqtt_port); client.setCallback(callback); } void loop() { // Lógica do Botão de Reset Físico (Pino 13) if (digitalRead(BOTAO_RESET_WIFI) == LOW) { if (tempoBotaoPressionado == 0) tempoBotaoPressionado = millis(); else if (millis() - tempoBotaoPressionado > 5000) { WiFiManager wm; wm.resetSettings(); ESP.restart(); } } else { tempoBotaoPressionado = 0; } if (!client.connected()) reconnect(); client.loop(); // Lógica dos Sensores e Alerta (RELE 15) bool aberto = (digitalRead(SENSOR_ABERTO_PIN) == LOW); bool fechado = (digitalRead(SENSOR_FECHADO_PIN) == LOW); if (aberto) { if (ultimoEstado != "ABERTO" && ultimoEstado != "ALERTA") { client.publish(topic_sensores, "{\"sensor2\":\"ABERTO\"}", true); ultimoEstado = "ABERTO"; tempoInicioAberto = millis(); monitorandoAlerta = true; } if (monitorandoAlerta && (millis() - tempoInicioAberto >= LIMITE_ALERTA)) { if (ultimoEstado != "ALERTA") { client.publish(topic_sensores, "{\"sensor2\":\"ALERTA\"}", true); ultimoEstado = "ALERTA"; digitalWrite(RELE_ALERTA_PIN, LOW); } } } else { digitalWrite(RELE_ALERTA_PIN, HIGH); monitorandoAlerta = false; if (fechado) { if (ultimoEstado != "FECHADO") { client.publish(topic_sensores, "{\"sensor2\":\"FECHADO\"}", true); ultimoEstado = "FECHADO"; } } else { if (ultimoEstado != "MOVIMENTO") { client.publish(topic_sensores, "{\"sensor2\":\"MOVIMENTO\"}", true); ultimoEstado = "MOVIMENTO"; } } } if (relePulsoAtivo && (millis() - tempoPulso >= 3000)) { digitalWrite(RELE_PIN, HIGH); relePulsoAtivo = false; } }
